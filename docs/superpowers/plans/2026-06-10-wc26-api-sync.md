# WC26 API Live Sync Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Automatically sync WC26 live scores + goal scorers from RapidAPI into Supabase every minute, display a pulsing "EN DIRECT" badge on live match cards, and give admins a manual sync button.

**Architecture:** A Supabase Edge Function (`sync-wc26`) fetches all matches from the API, maps them to Supabase rows (by team names for group matches, by id for KO matches 73-104), and upserts `result_home`, `result_away`, `live_status`, and `goals_home`/`goals_away`. A `pg_cron` job calls it every minute via `pg_net`. The existing Supabase Realtime subscription already propagates any DB update to all connected clients instantly. A new `live_status` column on `matches` drives the badge on the frontend.

**Tech Stack:** Supabase Edge Functions (Deno/TypeScript), pg_cron, pg_net, RapidAPI WC26 API (`wc26-live-football-api.p.rapidapi.com`), Vue 3 + TypeScript + Pinia

import { createClient } from 'https://esm.sh/@supabase/supabase-js@2'

interface ApiMatch {
  id: number
  home: string
  away: string
  group: string | null
  score: string | { home: number; away: number } | null
  status: string   // "scheduled" | "live" | "finished"
  minute: number
  played: boolean
  live_data: unknown
}

interface Goal {
  name: string
  minute: number
  penalty?: boolean
  owngoal?: boolean
}

function parseScore(score: ApiMatch['score']): { home: number; away: number } | null {
  if (!score) return null
  if (typeof score === 'object' && 'home' in score) return score
  if (typeof score === 'string') {
    const m = score.match(/(\d+)\s*[-:]\s*(\d+)/)
    if (m) return { home: parseInt(m[1]), away: parseInt(m[2]) }
  }
  return null
}

function parseGoals(liveData: unknown, side: 'home' | 'away'): Goal[] {
  if (!liveData || typeof liveData !== 'object') return []
  const d = liveData as Record<string, unknown>

  // Format A: { goals: [{team: "home", player: "Name", minute: 32, type: "goal"}] }
  if (Array.isArray(d.goals)) {
    return (d.goals as Record<string, unknown>[])
      .filter(g => g.team === side)
      .map(g => ({
        name: String(g.player ?? g.name ?? ''),
        minute: Number(g.minute ?? 0),
        penalty: g.type === 'penalty' || g.penalty === true,
        owngoal: g.type === 'own_goal' || g.owngoal === true,
      }))
  }

  // Format B: { goals_home: [{player, minute}], goals_away: [...] }
  const key = side === 'home' ? 'goals_home' : 'goals_away'
  if (Array.isArray(d[key])) {
    return (d[key] as Record<string, unknown>[]).map(g => ({
      name: String(g.player ?? g.name ?? ''),
      minute: Number(g.minute ?? 0),
      penalty: g.penalty === true,
      owngoal: g.owngoal === true,
    }))
  }

  return []
}

serve(async () => {
  const supabase = createClient(
    Deno.env.get('SUPABASE_URL')!,
    Deno.env.get('SUPABASE_SERVICE_ROLE_KEY')!,
  )
  const apiKey = Deno.env.get('RAPIDAPI_KEY')!

  // Fetch all matches from the API
  const apiRes = await fetch('https://wc26-live-football-api.p.rapidapi.com/matches', {
    headers: {
      'x-rapidapi-host': 'wc26-live-football-api.p.rapidapi.com',
      'x-rapidapi-key': apiKey,
    },
  })
  if (!apiRes.ok) {
    return new Response(JSON.stringify({ error: 'API fetch failed', status: apiRes.status }), { status: 502 })
  }
  const { data: apiMatches }: { data: ApiMatch[] } = await apiRes.json()

  // Fetch our DB matches to build lookup maps
  const { data: dbMatches, error: dbErr } = await supabase
    .from('matches')
    .select('id, home_team, away_team, live_status, result_home, result_away')
  if (dbErr) return new Response(JSON.stringify({ error: dbErr.message }), { status: 500 })

  // Build a (sorted team pair) → db row map for group matches
  const pairMap = new Map<string, typeof dbMatches[0]>()
  for (const m of dbMatches ?? []) {
    if (m.home_team && m.away_team) {
      pairMap.set([m.home_team, m.away_team].sort().join('|'), m)
    }
  }

  const upserts: Record<string, unknown>[] = []

  for (const am of apiMatches) {
    if (am.status === 'scheduled' && !am.played) continue  // nothing to update

    const frHome = TEAM_EN_FR[am.home] ?? am.home
    const frAway = TEAM_EN_FR[am.away] ?? am.away

    // Find DB match: for KO (id >= 73) use ID directly; for group matches use team pair
    let dbMatch: typeof dbMatches[0] | undefined
    if (am.id >= 73) {
      dbMatch = (dbMatches ?? []).find(m => m.id === am.id)
    } else {
      dbMatch = pairMap.get([frHome, frAway].sort().join('|'))
    }
    if (!dbMatch) continue

    const scoreObj = parseScore(am.score)
    const upsert: Record<string, unknown> = {
      id: dbMatch.id,
      live_status: am.status,
    }

    if (scoreObj) {
      // Determine if home/away are flipped (API home ≠ DB home_team)
      const flipped = dbMatch.home_team !== frHome && dbMatch.away_team === frHome
      upsert.result_home = flipped ? scoreObj.away : scoreObj.home
      upsert.result_away = flipped ? scoreObj.home : scoreObj.away

      if (am.live_data) {
        const apiGoalsHome = parseGoals(am.live_data, flipped ? 'away' : 'home')
        const apiGoalsAway = parseGoals(am.live_data, flipped ? 'home' : 'away')
        if (apiGoalsHome.length || apiGoalsAway.length) {
          upsert.goals_home = apiGoalsHome
          upsert.goals_away = apiGoalsAway
        }
      }
    }

    upserts.push(upsert)
  }

  if (upserts.length === 0) {
    return new Response(JSON.stringify({ synced: 0, message: 'No matches to update' }), { status: 200 })
  }

  const { error: upsertErr } = await supabase
    .from('matches')
    .upsert(upserts, { onConflict: 'id' })
  if (upsertErr) return new Response(JSON.stringify({ error: upsertErr.message }), { status: 500 })

  return new Response(JSON.stringify({ synced: upserts.length }), { status: 200 })
})
```

const syncMsg     = ref('')

async function syncFromApi() {
  syncLoading.value = true
  syncMsg.value = ''
  try {
    const { data, error } = await sb.functions.invoke('sync-wc26')
    if (error) throw error
    syncMsg.value = `✓ ${(data as { synced: number }).synced} match(s) synchronisé(s)`
  } catch (e) {
    syncMsg.value = '✗ ' + (e as Error).message
  } finally {
    syncLoading.value = false
    setTimeout(() => { syncMsg.value = '' }, 3000)
  }
}
```
